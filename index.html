birthdayBtn.addEventListener("click", () => { // Hide other content messageBox.style.display = "none"; document.getElementById("puzzle-box").style.display = "none"; memoriesBox.style.display = "none"; // Toggle birthday tab if (birthdayBox.style.display === "none" || birthdayBox.style.display === "") { birthdayBox.style.display = "block"; // Only create canvas once if (!document.getElementById("matrixCanvas")) { const canvas = document.createElement("canvas"); canvas.id = "matrixCanvas"; birthdayBox.appendChild(canvas); const ctx = canvas.getContext("2d"); canvas.width = birthdayBox.offsetWidth; canvas.height = 400; const words = ["I love you", "Zodwa Mthombeni", "Happy birthday", "You are the best"]; const fontSize = 24; const columns = Math.floor(canvas.width / fontSize); const drops = Array(columns).fill(0); function drawMatrix() { // Pink background ctx.fillStyle = "rgba(255,230,236,0.1)"; // light pink with slight trail ctx.fillRect(0, 0, canvas.width, canvas.height); // Red words ctx.fillStyle = "#ff4d6d"; ctx.font = fontSize + "px monospace"; for (let i = 0; i < drops.length; i++) { const text = words[Math.floor(Math.random() * words.length)]; ctx.fillText(text, i * fontSize, drops[i] * fontSize); if (drops[i] * fontSize > canvas.height && Math.random() > 0.975) { drops[i] = 0; } drops[i]++; } } setInterval(drawMatrix, 50); // Adjust canvas on window resize window.addEventListener("resize", () => { canvas.width = birthdayBox.offsetWidth; canvas.height = 400; }); } } else { birthdayBox.style.display = "none"; } });
